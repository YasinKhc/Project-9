Build a Model

## Lets split the data into three subsets, one for training, one for validation, and one for testing. We assign 50% to training set, 25% to validation and testing sets

train_samples_indices <- round(nrow(final_dataset_residential) * 0.5)

test_samples_indices <-
  nrow(final_dataset_residential) - train_samples_indices - valid_samples_indices

train_ds <- final_dataset_residential[seq(train_samples_indices),]

valid_ds <-
  final_dataset_residential[seq(from = train_samples_indices + 1,
                    length.out = valid_samples_indices),]

test_ds <-
  final_dataset_residential[seq(to = nrow(final_dataset_residential), length.out = test_samples_indices),]

## Lets use the data for the past 7 days, sampled once per hour, to predict the residential electricity usage in the next hour

ds_to_inputs_and_targets <- function(ds) {
  inputs <- ds %>% as.matrix()
  targets <- as.array(ds$Total_Hourly_Usage_Residential)
  list(
    head(inputs, -168),
    tail(targets, -168)
    )
}

sampling_rate = 1

sequence_length <- 168

batch_size <- 256

make_dataset <- function(ds) {
  c(inputs, targets) %<-% ds_to_inputs_and_targets(ds)
  ## timeseries_dataset_from_array takes care of time-series data preparation
  timeseries_dataset_from_array(
    inputs,
    targets,
    sampling_rate = sampling_rate,
    sequence_length = sequence_length,
    shuffle = TRUE,
    batch_size = batch_size
  )
}

train_dataset <- make_dataset(train_ds)
valid_dataset <- make_dataset(valid_ds)
test_dataset <- make_dataset(test_ds)

inputs <-
  layer_input(shape = c(sequence_length, ncol(final_dataset_residential)))

## I am using layer_gru() recurrent layer available in Keras. Feel free to use layer_simple_rnn() or layer_lstm()

outputs <- inputs %>%
  layer_gru(32,
            recurrent_dropout = 0.5,
            return_sequences = TRUE) %>%
  layer_gru(12, recurrent_dropout = 0.5) %>%
  layer_dropout(0.5) %>%
  layer_dense(1)

model <- keras_model(inputs, outputs)

## Keep the best model seen during training

callbacks <- list(
  callback_model_checkpoint("gru_dropout.keras",
                            save_best_only = TRUE)
)

model %>% compile(
  optimizer = optimizer_rmsprop(learning_rate = 0.001, decay = 1e-6),
  loss = "mae",
  metrics = "mae"
)
